# 增长实验管理系统
class GrowthExperimentEngine:
    """
    智链平台增长实验引擎
    基于ICE评分法和统计学原理的增长实验管理系统
    """
    
    def __init__(self):
        self.experiment_queue = []
        self.running_experiments = {}
        self.completed_experiments = []
        self.growth_metrics = GrowthMetrics()
        
    def evaluate_experiment_idea(self, idea: Dict) -> Dict:
        """
        使用ICE框架评估实验想法
        Impact(影响): 对关键指标的潜在影响 (1-10分)
        Confidence(信心): 对实验成功的信心程度 (1-10分)
        Ease(容易度): 实施实验的难易程度 (1-10分)
        """
        ice_score = {
            'impact': self.calculate_impact_score(idea),
            'confidence': self.calculate_confidence_score(idea),
            'ease': self.calculate_ease_score(idea),
        }
        
        total_score = (ice_score['impact'] * ice_score['confidence'] * ice_score['ease']) ** (1/3)
        
        return {
            'idea_id': idea['id'],
            'ice_scores': ice_score,
            'total_score': total_score,
            'priority_level': self.get_priority_level(total_score),
            'recommended_action': self.get_action_recommendation(total_score)
        }
    
    def create_ab_test(self, experiment_config: Dict) -> str:
        """
        创建A/B测试实验
        """
        experiment = {
            'id': self.generate_experiment_id(),
            'name': experiment_config['name'],
            'hypothesis': experiment_config['hypothesis'],
            'control_group': experiment_config['control'],
            'treatment_group': experiment_config['treatment'],
            'success_metrics': experiment_config['metrics'],
            'traffic_split': experiment_config.get('split', 0.5),
            'min_sample_size': self.calculate_sample_size(
                experiment_config['expected_effect'],
                experiment_config['significance_level'],
                experiment_config['power']
            ),
            'status': 'draft',
            'start_date': None,
            'end_date': None,
            'results': None
        }
        
        self.experiment_queue.append(experiment)
        return experiment['id']
    
    def run_cohort_analysis(self, user_segment: str, time_period: str) -> Dict:
        """
        运行用户群组分析
        """
        cohort_data = self.growth_metrics.get_cohort_data(user_segment, time_period)
        
        analysis_results = {
            'retention_curves': self.analyze_retention_curves(cohort_data),
            'ltv_trends': self.analyze_ltv_trends(cohort_data),
            'conversion_funnels': self.analyze_conversion_funnels(cohort_data),
            'behavioral_patterns': self.identify_behavioral_patterns(cohort_data),
            'churn_indicators': self.identify_churn_indicators(cohort_data)
        }
        
        return analysis_results

# 增长漏斗分析系统
class GrowthFunnelAnalyzer:
    """
    增长漏斗分析器
    """
    
    def __init__(self):
        self.funnel_stages = [
            'website_visit',
            'signup_start', 
            'signup_complete',
            'first_login',
            'onboarding_complete',
            'first_value_achieved',
            'active_user',
            'paid_conversion'
        ]
        
    def analyze_conversion_funnel(self, time_period: str) -> Dict:
        """
        分析转化漏斗各环节表现
        """
        funnel_data = {}
        
        for i, stage in enumerate(self.funnel_stages):
            stage_data = self.get_stage_metrics(stage, time_period)
            conversion_rate = self.calculate_stage_conversion_rate(stage, time_period)
            
            funnel_data[stage] = {
                'users': stage_data['user_count'],
                'conversion_rate': conversion_rate,
                'drop_off_rate': 1 - conversion_rate if i > 0 else 0,
                'avg_time_to_convert': stage_data['avg_conversion_time'],
                'top_drop_off_points': self.identify_drop_off_points(stage)
            }
            
        # 识别最大优化机会
        optimization_opportunities = self.identify_optimization_opportunities(funnel_data)
        
        return {
            'funnel_data': funnel_data,
            'overall_conversion': self.calculate_overall_conversion(funnel_data),
            'biggest_opportunities': optimization_opportunities,
            'recommended_experiments': self.recommend_experiments(optimization_opportunities)
        }
    
    def identify_optimization_opportunities(self, funnel_data: Dict) -> List[Dict]:
        """
        识别漏斗优化机会
        """
        opportunities = []
        
        for stage, data in funnel_data.items():
            if data['drop_off_rate'] > 0.5:  # 流失率超过50%
                impact_score = data['users'] * (1 - data['conversion_rate'])
                
                opportunities.append({
                    'stage': stage,
                    'current_conversion_rate': data['conversion_rate'],
                    'drop_off_users': impact_score,
                    'potential_impact': 'high' if impact_score > 1000 else 'medium',
                    'suggested_experiments': self.get_stage_experiments(stage)
                })
                
        return sorted(opportunities, key=lambda x: x['drop_off_users'], reverse=True)

// 客户健康度评分系统
class HealthScoreCalculator {
    constructor() {
        this.weights = {
            product_usage: 0.25,      // 产品使用频率和深度
            feature_adoption: 0.20,   // 功能采用率
            support_engagement: 0.15, // 客服互动质量
            business_outcomes: 0.20,  // 业务成果实现
            relationship_health: 0.10, // 关系健康度
            account_growth: 0.10      // 账户增长情况
        };
    }
    
    calculate(customer) {
        const scores = {
            product_usage: this.calculateUsageScore(customer),
            feature_adoption: this.calculateAdoptionScore(customer),
            support_engagement: this.calculateSupportScore(customer),
            business_outcomes: this.calculateOutcomesScore(customer),
            relationship_health: this.calculateRelationshipScore(customer),
            account_growth: this.calculateGrowthScore(customer)
        };
        
        // 加权计算总分
        const totalScore = Object.keys(scores).reduce((total, key) => {
            return total + (scores[key] * this.weights[key]);
        }, 0);
        
        return {
            total_score: Math.round(totalScore),
            component_scores: scores,
            health_trend: this.calculateTrend(customer),
            risk_factors: this.identifyRiskFactors(scores),
            improvement_opportunities: this.identifyOpportunities(scores)
        };
    }
    
    calculateUsageScore(customer) {
        const usage = customer.usage_metrics;
        let score = 0;
        
        // 登录频率 (40%)
        const loginFrequency = usage.avg_logins_per_week;
        score += Math.min(loginFrequency / 5, 1) * 40;
        
        // 功能使用深度 (30%)
        const featureUsage = usage.unique_features_used / usage.total_available_features;
        score += featureUsage * 30;
        
        // 会话时长 (30%)
        const sessionLength = usage.avg_session_duration_minutes;
        score += Math.min(sessionLength / 30, 1) * 30;
        
        return Math.round(score);
    }
    
    calculateBusinessValue(customer) {
        const outcomes = customer.business_outcomes;
        
        // 计算ROI实现率
        const roiAchievement = outcomes.actual_roi / outcomes.expected_roi;
        
        // 计算效率提升
        const efficiencyGain = outcomes.efficiency_improvement_percentage;
        
        // 计算成本节约
        const costSavings = outcomes.cost_savings_amount;
        
        return {
            roi_achievement: Math.min(roiAchievement, 2), // 最高200%
            efficiency_gain: efficiencyGain,
            cost_savings: costSavings,
            overall_value_score: this.calculateOverallValueScore([
                roiAchievement, 
                efficiencyGain / 100, 
                Math.min(costSavings / 100000, 1)
            ])
        };
    }
}
