## 🎯 布局设计新原则

### 🛡️ 安全导向的导览式设计

基于安全性和页面加载性能考虑，采用**导览页面**设计模式：
- **主页**: 纯导览功能，无复杂业务逻辑
- **动画展示**: 科技感动画框，吸引用户点击
- **快速入口**: 清晰的功能区域划分
- **身份识别**: 右侧登录区域，支持多平台认证

// 聊天消息组件
interface ChatMessageProps {
  message: {
    id: string;
    type: 'user' | 'agent';
    agent?: AgentRole;
    content: string;
    timestamp: Date;
    status: 'sending' | 'sent' | 'analyzing';
  };
}

const ChatMessage: React.FC<ChatMessageProps> = ({ message }) => {
  const isAgent = message.type === 'agent';
  
  return (
    <div className={cn(
      "flex gap-4 p-6",
      isAgent ? "bg-cloudsway-glass-bg" : "bg-transparent"
    )}>
      {/* 头像 */}
      {isAgent && message.agent && (
        <div className={`agent-avatar agent-${message.agent}`}>
          {AGENT_ICONS[message.agent]}
        </div>
      )}
      
      {/* 消息内容 */}
      <div className="flex-1">
        {isAgent && (
          <div className="flex items-center gap-2 mb-2">
            <span className="font-semibold text-sm">
              {AGENTS[message.agent!].name}
            </span>
            <span className="text-xs text-text-muted">
              {AGENTS[message.agent!].role}
            </span>
          </div>
        )}
        
        <div className={cn(
          "prose prose-sm max-w-none",
          isAgent ? "text-text-primary" : "text-text-secondary"
        )}>
          <ReactMarkdown>{message.content}</ReactMarkdown>
        </div>
        
        <div className="flex items-center justify-between mt-3">
          <time className="text-xs text-text-muted">
            {formatTime(message.timestamp)}
          </time>
          
          {message.status === 'analyzing' && (
            <div className="flex items-center gap-2 text-xs text-cloudsway-primary-500">
              <div className="loading-rainbow w-4 h-4 rounded-full" />
              分析中...
            </div>
          )}
        </div>
      </div>
      
      {!isAgent && (
        <div className="w-8 h-8 bg-cloudsway-primary-500 rounded-full flex items-center justify-center">
          <User className="w-4 h-4 text-white" />
        </div>
      )}
    </div>
  );
};
