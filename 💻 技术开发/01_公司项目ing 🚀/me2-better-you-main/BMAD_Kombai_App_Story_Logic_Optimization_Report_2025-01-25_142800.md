          <p className="text-xl text-gray-600">
            不是简单的AI助手，而是完整的专业决策支持系统
          </p>
        </div>
        
        {/* 交互式架构展示 */}
        <div className="grid grid-cols-1 md:grid-cols-3 gap-8 mb-16">
          {Object.entries(levels).map(([level, config]) => (
            <motion.div
              key={level}
              className={`p-8 rounded-2xl cursor-pointer transition-all duration-300 ${
                activeLevel === parseInt(level) 
                  ? `bg-${config.color} text-white shadow-2xl scale-105` 
                  : 'bg-gray-50 hover:bg-gray-100'
              }`}
              onClick={() => setActiveLevel(parseInt(level) as 1 | 2 | 3)}
              whileHover={{ y: -5 }}
              whileTap={{ scale: 0.98 }}
            >
              <div className="text-center mb-6">
                <div className={`w-16 h-16 mx-auto mb-4 rounded-full flex items-center justify-center ${
                  activeLevel === parseInt(level) ? 'bg-white/20' : `bg-${config.color}/10`
                }`}>
                  <span className="text-2xl font-bold">{level}</span>
                </div>
                <h3 className="text-xl font-bold mb-2">{config.title}</h3>
                <p className={`text-sm ${activeLevel === parseInt(level) ? 'text-white/80' : 'text-gray-600'}`}>
                  {config.subtitle}
                </p>
              </div>
              
              <div className="space-y-3">
                {config.capabilities.map((capability, index) => (
                  <motion.div
                    key={index}
                    className={`flex items-center space-x-2 ${
                      activeLevel === parseInt(level) ? 'text-white/90' : 'text-gray-700'
                    }`}
                    initial={{ opacity: 0, x: -20 }}
                    animate={{ opacity: 1, x: 0 }}
                    transition={{ duration: 0.3, delay: index * 0.1 }}
                  >
                    <CheckCircle2 size={16} />
                    <span className="text-sm">{capability}</span>
                  </motion.div>
                ))}
              </div>
            </motion.div>
          ))}
        </div>
        
        {/* 活跃层级详细展示 */}
        <motion.div
          key={activeLevel}
          className="bg-gray-50 rounded-2xl p-8"
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 0.5 }}
        >
          <div className="grid grid-cols-1 md:grid-cols-2 gap-8 items-center">
            <div>
              <h4 className="text-2xl font-bold mb-4">
                {levels[activeLevel].description}
              </h4>
              <p className="text-gray-600 mb-6">
                深入展示{levels[activeLevel].title}如何为专业人士提供决策支持
              </p>
              <div className="space-y-4">
                {levels[activeLevel].capabilities.map((capability, index) => (
                  <div key={index} className="flex items-start space-x-3">
                    <div className={`w-6 h-6 rounded-full bg-${levels[activeLevel].color}/10 flex items-center justify-center flex-shrink-0 mt-0.5`}>
                      <ArrowRight size={12} className={`text-${levels[activeLevel].color}`} />
                    </div>
                    <span className="text-gray-700">{capability}</span>
                  </div>
                ))}
              </div>
            </div>
            
            {/* 可视化演示区域 */}
            <div className="relative">
              <LevelDemonstration level={activeLevel} />
            </div>
          </div>
        </motion.div>
      </div>
    </section>
  );
};
