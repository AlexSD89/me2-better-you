## 🎯 第一性原理分析

### 核心价值定位
- **使命**: 建立AI产品信任基础设施
- **目标用户**: 企业CTO、采购经理、技术负责人
- **转化路径**: 专业可信度 → 试用体验 → 付费决策
- **关键差异**: 从"评测工具"升级为"信任服务生态"

### 用户痛点与设计目标

| 用户角色 | 核心痛点 | 设计目标 |
|---------|---------|---------|
| **企业CTO** | AI选型信息不对称，技术风险难评估 | 专业技术展示，降低认知门槛 |
| **采购经理** | 成本效益难量化，供应商可靠性存疑 | ROI可视化，权威认证展示 |
| **技术负责人** | 实施复杂度未知，兼容性风险高 | 详细技术参数，实施指南 |

### 3秒信任建立法则

**黄金3秒内必须传达的信任信号**:
1. **权威认证** - AI-China-1标识、政府合规展示
2. **专业数据** - 真实的评测数量、客户案例
3. **专家背书** - 前大厂AI专家团队展示
4. **成功案例** - 知名企业客户logo墙

### 评测雷达图 (Radar Chart)
```tsx
// 多维度评测雷达图组件
interface RadarChartProps {
  data: {
    performance: number;    // 技术性能
    security: number;       // 安全合规
    cost: number;          // 成本效益
    usability: number;     // 易用性
    reliability: number;   // 可靠性
    scalability: number;   // 可扩展性
  };
  size?: 'sm' | 'md' | 'lg';
  showLabels?: boolean;
}

// 设计原则
const radarChartDesign = {
  colors: {
    primary: '#0066CC',      // 智评蓝
    secondary: '#00CC66',    // 智评绿
    grid: '#E5E7EB',        // 网格线
    text: '#6B7280',        // 标签文字
  },
  gradients: {
    fill: 'rgba(0, 102, 204, 0.1)', // 半透明填充
    stroke: '#0066CC',               // 边框色
  },
  animation: {
    duration: 800,           // 动画时长
    easing: 'ease-out',     // 缓动函数
  }
};
```

```tsx
        className="absolute left-0 top-0 h-full bg-gradient-to-r from-zhiping-blue-500 to-zhiping-green-500 rounded-full transition-all duration-500"
```

```tsx
  const [count, setCount] = useState(0);

      const timer = setInterval(() => {

          return value;
        });
      }, 50);
    }
```

```tsx
// 内容懒加载组件
export function LazySection({ 
  children, 
  threshold = 0.1,
  rootMargin = '50px'
}: {
  children: React.ReactNode;
  threshold?: number;
  rootMargin?: string;
}) {
  const [isVisible, setIsVisible] = useState(false);
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsVisible(true);
          observer.disconnect();
        }
      },
      { threshold, rootMargin }
    );

    if (ref.current) {
      observer.observe(ref.current);
    }

    return () => observer.disconnect();
  }, [threshold, rootMargin]);

  return (
    <div ref={ref} className="min-h-32">
      {isVisible ? children : (
        <div className="flex items-center justify-center h-32">
          <Loader2 className="h-6 w-6 animate-spin text-zhiping-blue-500" />
        </div>
      )}
    </div>
  );
}
```

### 键盘导航支持
```tsx
// 键盘可访问的导航菜单
export function AccessibleNavigation() {
  const [activeIndex, setActiveIndex] = useState(0);
  const menuItems = useRef<HTMLElement[]>([]);

  const handleKeyDown = (event: KeyboardEvent) => {
    switch (event.key) {
      case 'ArrowDown':
        event.preventDefault();
        setActiveIndex((prev) => (prev + 1) % menuItems.current.length);
        break;
      case 'ArrowUp':
        event.preventDefault();
        setActiveIndex((prev) => (prev - 1 + menuItems.current.length) % menuItems.current.length);
        break;
      case 'Enter':
      case ' ':
        event.preventDefault();
        menuItems.current[activeIndex]?.click();
        break;
      case 'Escape':
        setActiveIndex(-1);
        break;
    }
  };

  return (
    <nav role="navigation" aria-label="主导航" onKeyDown={handleKeyDown}>
      {navigationItems.map((item, index) => (
        <Link
          key={item.href}
          href={item.href}
          ref={(el) => el && (menuItems.current[index] = el)}
          className={`nav-item ${activeIndex === index ? 'nav-item-active' : ''}`}
          aria-current={activeIndex === index ? 'page' : undefined}
          tabIndex={0}
        >
          {item.label}
        </Link>
      ))}
    </nav>
  );
}
```
